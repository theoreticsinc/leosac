72 bool StdinControllerModule::send_request(std::shared_ptr<zmqpp::socket> target,
73  const std::vector<std::string> &cmds)
74 {
75  zmqpp::message msg;

77  for (const auto &str : cmds)
78  {

84  msg << nbr;

86  else

90  }

92

96  if (!p.poll(1000))
