64 void InstrumentationModule::handle_gpio_command(zmqpp::message *str)
65 {
66  assert(str);

69 
70  *str >> gpio_name >> cmd;
71  DEBUG("GPIO CMD: " << gpio_name << ", " << cmd);
72  if (cmd == "ON")
73  {
74  bus_push_.send(zmqpp::message() << ("S_" + gpio_name) << "ON");

79  }
