79 void WiegandReaderModule::process_config()
80 {
81  boost::property_tree::ptree module_config = config_.get_child("module_config");
82 
83  if (module_config.get<bool>("use_database", false))
84  {
85  auto hwd_service =
86  get_service_registry().get_service<Hardware::HardwareService>();
87  ASSERT_LOG(hwd_service, "No hardware service but we have database.");
88  hwd_service->register_serializer<WiegandReaderConfig>(
89  &WiegandReaderConfigSerializer::serialize);
90  load_db_config();
91  }
92  else
93  {
94  load_xml_config(module_config);
95  }
96 
97  // Now we process the configuration object.
98  for (const auto &reader_config : wiegand_config_->readers())
99  {
100  using namespace Colorize;
101  INFO("Creating WiegandReader: "
102  << green(underline(reader_config->name())) << "\n\t Green Led: "
103  << green(underline(reader_config->green_led_name()))
104  << "\n\t Buzzer: " << green(underline(reader_config->buzzer_name()))
105  << "\n\t GPIO Low: " << green(underline(reader_config->gpio_low_name()))
106  << "\n\t GPIO High: "
107  << green(underline(reader_config->gpio_high_name()))
108  << "\n\t Mode: " << green(underline(reader_config->mode)));
109 
110  WiegandReaderImpl reader(
111  ctx_, reader_config->name(), reader_config->gpio_high_name(),
112  reader_config->gpio_low_name(), reader_config->green_led,
113  reader_config->buzzer, create_strategy(*reader_config, &reader));
114  utils_->config_checker().register_object(reader.name(),
115  ConfigChecker::ObjectType::READER);
116  readers_.push_back(std::move(reader));
117  }
118 }

120 void WiegandReaderModule::run()
121 {
122  if (config_.get_child("module_config").get<bool>("use_database", false))
123  {
124  ws_helper_thread_ = std::make_unique<WSHelperThread>(utils_);
125  ws_helper_thread_->start_running();
126  }
127  while (is_running_)
128  {
129  if (!reactor_.poll(50))
130  {
131  for (auto &reader : readers_)
132  reader.timeout();
133  }
134  }
135  auto ws_service = get_service_registry().get_service<WebSockAPI::Service>();
136  if (ws_service && ws_helper_thread_)
137  ws_helper_thread_->unregister_ws_handlers(*ws_service);
138 }

140 Strategy::WiegandStrategyUPtr
141 WiegandReaderModule::create_strategy(const WiegandReaderConfig &reader_cfg,
142  WiegandReaderImpl *reader)
143 {
144  using namespace Auth;
145  using namespace Strategy;
146 
147  auto simple_wiegand =
148  std::unique_ptr<CardReading>(new SimpleWiegandStrategy(reader));
149  auto pin_4bits = std::unique_ptr<PinReading>(new WiegandPinNBitsOnly<4>(
150  reader, reader_cfg.pin_timeout, reader_cfg.pin_key_end));
151  auto pin_8bits = std::unique_ptr<PinReading>(new WiegandPinNBitsOnly<8>(
152  reader, reader_cfg.pin_timeout, reader_cfg.pin_key_end));
153  auto pin_buffered = std::unique_ptr<PinReading>(new WiegandPinBuffered(reader));
154 
155  if (reader_cfg.mode == "SIMPLE_WIEGAND")
156  return std::move(simple_wiegand);
157  else if (reader_cfg.mode == "WIEGAND_PIN_4BITS")
158  return std::move(pin_4bits);
159  else if (reader_cfg.mode == "WIEGAND_PIN_8BITS")
160  return std::move(pin_8bits);
161  else if (reader_cfg.mode == "WIEGAND_PIN_BUFFERED")
162  return std::move(pin_buffered);
163  else if (reader_cfg.mode == "WIEGAND_CARD_PIN_4BITS")
164  {
165  return std::unique_ptr<WiegandStrategy>(
166  new WiegandCardAndPin(reader, std::move(simple_wiegand),
167  std::move(pin_4bits), reader_cfg.pin_timeout));
168  }
169  else if (reader_cfg.mode == "WIEGAND_CARD_PIN_8BITS")
170  {
171  return std::unique_ptr<WiegandStrategy>(
172  new WiegandCardAndPin(reader, std::move(simple_wiegand),
173  std::move(pin_8bits), reader_cfg.pin_timeout));
174  }
175  else if (reader_cfg.mode == "WIEGAND_CARD_PIN_BUFFERED")
176  {
177  return std::unique_ptr<WiegandStrategy>(
178  new WiegandCardAndPin(reader, std::move(simple_wiegand),
179  std::move(pin_buffered), reader_cfg.pin_timeout));
180  }
181  else if (reader_cfg.mode == "AUTODETECT")
182  {
183  return std::unique_ptr<WiegandStrategy>(
184  new Autodetect(reader, reader_cfg.pin_timeout, reader_cfg.pin_key_end));
185  }
186  else
187  {
188  ERROR("Wiegand mode " << reader_cfg.mode << " is not a valid mode.");
189  assert(0);
190  throw std::runtime_error("Invalid wiegand mode " + reader_cfg.mode);
191  }
192  return nullptr;
193 }

195 void WiegandReaderModule::load_db_config()
196 {
197  using namespace odb;
198  using namespace odb::core;
199  auto db = utils_->database();
200 
201  // First we load or update database schema if needed.
202  schema_version v = db->schema_version("module_wiegand");
203  schema_version cv(schema_catalog::current_version(*db, "module_wiegand"));
204  if (v == 0)
205  {
206  transaction t(db->begin());
207  INFO("Attempt to create module_wiegand SQL schema.");
208  schema_catalog::create_schema(*db, "module_wiegand");
209  t.commit();
210  }
211  else if (v < cv)
212  {
213  INFO("Wiegand Module performing database migration. Going from version "
214  << v << " to version " << cv);
215  transaction t(db->begin());
216  schema_catalog::migrate(*db, cv, "module_wiegand");
217  t.commit();
218  }
219 
220  // Create empty configuration object...
221  wiegand_config_ = std::make_unique<WiegandConfig>();
222 
223  // ... then loads reader from database.
224  odb::transaction t(utils_->database()->begin());
225  odb::result<WiegandReaderConfig> result(
226  utils_->database()->query<WiegandReaderConfig>());
227  for (const auto &reader : result)
228  {
229  WiegandReaderConfigPtr reader_ptr =
230  db->load<WiegandReaderConfig>(reader.id());
231  ASSERT_LOG(reader_ptr, "Loading from database/cache failed");
232  if (reader_ptr->enabled())
233  wiegand_config_->add_reader(reader_ptr);
234  }
235  t.commit();
236  INFO("Wiegand module using SQL database for configuration.");
237 }

239 void WiegandReaderModule::load_xml_config(
240  const boost::property_tree::ptree &module_config)
241 {
242  wiegand_config_ = std::make_unique<WiegandConfig>();
243  for (auto &node : module_config.get_child("readers"))
244  {
245  auto reader_config = std::make_shared<WiegandReaderConfig>();
246  boost::property_tree::ptree xml_reader_cfg = node.second;
247 
248  // For GPIO object, we instanciate a "dummy" GPIO object that would
249  // normally be fetched from the database. The only goal of the GPIO
250  // object is to hold the device's name
251 
252  auto gpio_high = std::make_shared<Hardware::GPIO>();
253  gpio_high->name(xml_reader_cfg.get_child("high").data());
254 
255  auto gpio_low = std::make_shared<Hardware::GPIO>();
256  gpio_low->name(xml_reader_cfg.get_child("low").data());
257 
258  reader_config->name(xml_reader_cfg.get_child("name").data());
259  reader_config->gpio_high_ = gpio_high;
260  reader_config->gpio_low_ = gpio_low;
261  reader_config->buzzer = xml_reader_cfg.get<std::string>("buzzer", "");
262  reader_config->green_led = xml_reader_cfg.get<std::string>("green_led", "");
263 
264  reader_config->mode =
265  xml_reader_cfg.get<std::string>("mode", "SIMPLE_WIEGAND");
266  reader_config->pin_timeout =
267  std::chrono::milliseconds(xml_reader_cfg.get<int>("pin_timeout", 2500));
268  reader_config->pin_key_end = xml_reader_cfg.get<char>("pin_key_end", '#');
269 
270  config_check(reader_config->gpio_low_name(),
271  ConfigChecker::ObjectType::GPIO);
272  config_check(reader_config->gpio_high_name(),
273  ConfigChecker::ObjectType::GPIO);
274 
275  if (!reader_config->green_led_name().empty())
276  config_check(reader_config->green_led_name(),
277  ConfigChecker::ObjectType::LED);
278  if (!reader_config->buzzer_name().empty())
279  config_check(reader_config->buzzer_name(),
280  ConfigChecker::ObjectType::BUZZER);
281 
282  wiegand_config_->add_reader(reader_config);
283  }
