34 json WiegandReaderConfigSerializer::serialize(const WiegandReaderConfig &in,
35  const SecurityContext &sc)
36 {
37  json serialized = Hardware::RFIDReaderSerializer::serialize(in, sc);
38  // Now we override the type.
39  ASSERT_LOG(serialized.at("type").is_string(),
40  "Base GPIO serialization did something unexpected.");
41 
42  serialized["type"] = "wiegand-reader";
43  serialized["attributes"]["green_led"] = in.green_led_name();
44  serialized["attributes"]["buzzer"] = in.buzzer_name();
45  serialized["attributes"]["mode"] = in.mode;
46 
47  serialized["relationships"] = json{};
48 
49  auto hwd_service =
50  get_service_registry().get_service<Hardware::HardwareService>();
51  ASSERT_LOG(hwd_service, "No hardware service.");
52 
53  serialized["relationships"]["gpio-high"] =
54  hwd_service->serialize_device_metadata(in.gpio_high_);
55  serialized["relationships"]["gpio-low"] =
56  hwd_service->serialize_device_metadata(in.gpio_low_);
57 
58  return serialized;
59 }

61 void WiegandReaderConfigSerializer::unserialize(WiegandReaderConfig &out,
62  const json &in,
63  const SecurityContext &sc)
64 {
65  using namespace JSONUtil;
66  Hardware::RFIDReaderSerializer::unserialize(out, in, sc);
67 
68  out.mode = extract_with_default(in, "mode", out.mode);
69  out.green_led = extract_with_default(in, "green_led", out.green_led_name());
70  out.buzzer = extract_with_default(in, "buzzer", out.buzzer_name());
71 
72  // High GPIO
73  Hardware::DeviceId gpio_id = extract_with_default(in, "gpio_high_id", UUID{});
74  auto db = get_service_registry().get_service<DBService>();
75  if (!gpio_id.is_nil())
76  {
77  out.gpio_high_ =
78  odb::lazy_shared_ptr<Hardware::GPIO>(*db->db(), gpio_id).load();
79  }
80 
81  // Low GPIO
82  gpio_id = extract_with_default(in, "gpio_low_id", UUID{});
83  if (!gpio_id.is_nil())
84  {
85  out.gpio_low_ =
86  odb::lazy_shared_ptr<Hardware::GPIO>(*db->db(), gpio_id).load();
87  }
88 }
89 }
