60  template <typename HandlerT>
61  bool register_asio_handler(HandlerT &&handler, const std::string &type,
62  boost::asio::io_service &io)
63  {
64  // We want the `handler` callable to be executed in the current
65  // thread/io_service.
66 
67  // We wrap the user handler into a packaged task that will be `post()`ed
68  // onto the io_service (`io`) queue.
69 
70  WSHandler wrapped_handler = [&, handler](const RequestContext &req_ctx) {
71  std::packaged_task<boost::optional<json>(const RequestContext &)> pt(
72  [&](const RequestContext &req_ctx) { return handler(req_ctx); });
73  auto future = pt.get_future();
74  io.post([&]() { pt(req_ctx); });
75  return future.get();
76  };
77  return register_typed_handler(wrapped_handler, type);
78  }
