46  zmqpp::context &zmq_ctx, CoreUtilsPtr utils)
47 {
48  try
49  {
50  UserModule module(zmq_ctx, pipe, cfg, utils);
51  INFO("Module " << get_module_name()

53  pipe->send(zmqpp::signal::ok);

57  catch (const std::exception &e)

59  ERROR("An exception occurred in module code. Leosac will terminate."
