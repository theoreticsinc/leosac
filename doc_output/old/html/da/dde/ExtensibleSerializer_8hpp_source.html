45 template <typename SerializedT, typename ObjectT, typename... AdditionalArgs>
46 class ExtensibleSerializer
47 {
48  public:
49  SerializedT serialize(const ObjectT &input, AdditionalArgs &&... args)
50  {
51  std::lock_guard<std::mutex> lg(mutex_);
52 
53  auto type_index = boost::typeindex::type_id_runtime(input);
54  auto itr = serializers_.find(type_index);
55  if (itr != serializers_.end())
56  {
57  // Invoke the adapter we stored in the map.
58  // The wrapper will invoke the user-defined callable.
59  return itr->second(input, std::forward<AdditionalArgs>(args)...);
60  }
61  ASSERT_LOG(false, "Cannot find an appropriate serializer for " +
62  type_index.pretty_name());
63  return {};
64  }
65 
69  using SerializationCallable =
70  std::function<SerializedT(const ObjectT &, AdditionalArgs &&...)>;
71 
75  template <typename T, typename Callable>
76  void register_serializer(Callable &&callable)
77  {
78  static_assert(std::is_base_of<ObjectT, T>::value,
79  "T is not a subclass of ObjectT");
80 
81  std::lock_guard<std::mutex> lg(mutex_);
82  auto type_index = boost::typeindex::type_id<T>();
83  ASSERT_LOG(serializers_.count(type_index) == 0,
84  "Already got a serializer for this type of object.");
85 
86  // This is an adapter that wraps the user function. This allows us to
87  // store uniform function objects.
88  auto wrapper = [callable](const ObjectT &base_obj,
89  AdditionalArgs &&... args) -> SerializedT {
90  // This cast is safe, because we'll only pick this serializer when
91  // the typeid of the base type (ObjectT) match T.
92  const T &concrete_object = reinterpret_cast<const T &>(base_obj);
93 
94  // Extra runtime safety check.
95  const T &extra_safety = assert_cast<const T &>(base_obj);
96  ASSERT_LOG(std::addressof(concrete_object) ==
97  std::addressof(extra_safety),
98  "Memory error.");
99  return callable(concrete_object, std::forward<AdditionalArgs>(args)...);
100  };
101  serializers_[type_index] = wrapper;
102  };
103 
104  template <typename T>
105  void unregister_serializer()
106  {
107  std::lock_guard<std::mutex> lg(mutex_);
108 
109  auto type_index = boost::typeindex::type_id<T>();
110  serializers_.erase(type_index);
111  }
112 
113  private:
114  mutable std::mutex mutex_;
115  std::map<boost::typeindex::type_index, SerializationCallable> serializers_;
